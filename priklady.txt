MODULE e3;

CONST
    factorial = 10;

PROCEDURE recFact(f: INTEGER): INTEGER;
BEGIN
    IF f <= 1 THEN
        RETURN 1;
    ELSE
        RETURN f * recFact(f - 1);
    END;
END recFact;

BEGIN
    PRINT_STR("Factorial: ");
    PRINT_INT(recFact(factorial));
END e3.
